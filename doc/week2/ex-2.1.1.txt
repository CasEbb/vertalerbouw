latexTabular ::= beginTabular colsSpec rows endTabular
colsSpec     ::= LCURLY identiﬁer RCURLY
rows         ::= row* | e
row          ::= (entry (AMPERSAND entry)* | e) DOUBLE_BSLASH
entry        ::= num | identiﬁer | e
beginTabular ::= BSLASH BEGIN LCURLY TABULAR RCURLY
endTabular   ::= BSLASH END LCURLY TABULAR RCURLY
num          ::= digit+
identiﬁer    ::= letter (letter|digit)*
digit        ::= 0-9
letter       ::= A-z
BSLASH       ::= "\"
DOUBLE_BLASH ::= "\\"
LCURLY       ::= "{"
RCURLY       ::= "}"
AMPERSAND    ::= "&"
BEGIN        ::= "begin"
END          ::= "end"
TABULAR      ::= "tabular"
